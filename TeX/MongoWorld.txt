    \subsection{Nested array}
    \paragraph{}
    Finding the length or number of elements in an array which is nested inside a document
    \begin{lstlisting}[language=Python]
    db.listening_history.aggregate(
        ... { $project : { songs : 1  }  },
        ... { $unwind : "$songs"  },
        ... { $group: { _id: "result", count: { $sum: 1  } } }
        )
    { "_id" : "result", "count" : 83  }
    \end{lstlisting}
